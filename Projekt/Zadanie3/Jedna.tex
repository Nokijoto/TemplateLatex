\fs{12}

CASE
	When Wypłata<5000 THEN 'Zarabia mało'
	WHEN Wypłata>5000 THEN 'Zarabia średnio'
	WHEN Wypłata>10000 THEN 'Zarabia dużo'

CONCAT(lower(imie),'-',LOWER(nazwisko),'@hospital.com') as Email 
from Pracownicy
